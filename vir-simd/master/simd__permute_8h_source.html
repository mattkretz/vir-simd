28 template <typename F>
29 concept index_permutation_function_nosize = requires(F const& f)
30 {
31 { f(vir::cw<0>) } -> std::integral;
32#if __GNUC__ >= 15
33 typename std::integral_constant<int, f(vir::cw<0>)>;

37 template <typename F, std::size_t Size>
38 concept index_permutation_function_size = requires(F const& f)
39 {
40 { f(vir::cw<0>, vir::cw<Size>) } -> std::integral;
41#if __GNUC__ >= 15
42 typename std::integral_constant<int, f(vir::cw<0>, vir::cw<Size>)>;
43#endif
44 };

46 template <typename F, std::size_t Size>
47 concept index_permutation_function
48 = index_permutation_function_size<F, Size> or index_permutation_function_nosize<F>;

90 operator()(unsigned i, auto size) const
91 {
92 static_assert(size % (2 * N) == 0,
93 "swap_neighbors<N> permutation requires a multiple of 2N elements");

177 template <std::size_t N = 0, vir::any_simd_or_mask V,
178 detail::index_permutation_function<V::size()> F>
179 VIR_ALWAYS_INLINE constexpr stdx::resize_simd_t<N == 0 ? V::size() : N, V>
180 simd_permute(V const& v, F const idx_perm) noexcept
181 {
182 using T = typename V::value_type;
183 using R = stdx::resize_simd_t<N == 0 ? V::size() : N, V>;
184
185#if defined __GNUC__
186 if (not std::is_constant_evaluated())
187 if constexpr (std::has_single_bit(sizeof(V)) and V::size() <= stdx::native_simd<T>::size())
188 {
189#if defined __AVX2__
190 using v4df [[gnu::vector_size(32)]] = double;
191 if constexpr (std::same_as<T, float> and std::is_trivially_copyable_v<V>
192 and sizeof(v4df) == sizeof(V)
193 and requires {
194 F::is_even_rotation;
195 F::Offset;
196 { std::bool_constant<F::is_even_rotation>() }
197 -> std::same_as<std::true_type>;
198 })
199 {
200 const v4df intrin = detail::bit_cast<v4df>(v);
201 constexpr int control = ((F::Offset / 2) << 0)
202 | (((F::Offset / 2 + 1) % 4) << 2)
203 | (((F::Offset / 2 + 2) % 4) << 4)
204 | (((F::Offset / 2 + 3) % 4) << 6);
205 return detail::bit_cast<R>(__builtin_ia32_permdf256(intrin, control));
206 }
207#endif
208#if VIR_HAVE_WORKING_SHUFFLEVECTOR
209 if constexpr (std::has_single_bit(sizeof(V)) and std::has_single_bit(sizeof(R)))
210 {
211 using VBuiltin [[gnu::vector_size(sizeof(V))]] = T;
212 using RBuiltin [[gnu::vector_size(sizeof(R))]] = T;
213 if constexpr (std::is_trivially_copyable_v<V> and std::is_trivially_copyable_v<R>
214 and sizeof(VBuiltin) == sizeof(V) and sizeof(RBuiltin) == sizeof(R))
215 {
216 const VBuiltin vec = detail::bit_cast<VBuiltin>(v);
217 constexpr auto idx_perm2 = [=](constexpr_value auto i) {
218 if constexpr (detail::index_permutation_function_nosize<F>)
219 return vir::cw<idx_perm(i)>;
220 else
221 return vir::cw<idx_perm(i, vir::cw<V::size()>)>;
222 };
223 constexpr auto adj_idx = [](constexpr_value auto i) {
224 constexpr int j = i;
225 if constexpr (j == simd_permute_zero)
226 return vir::cw<V::size()>;
227 else if constexpr (j == simd_permute_uninit)
228 return vir::cw<-1>;
229 else if constexpr (j < 0)
230 {
231 static_assert (-j <= int(V::size()));
232 return vir::cw<int(V::size()) + j>;
233 }
234 else
235 {
236 static_assert (j < int(V::size()));
237 return vir::cw<j>;
238 }
239 };
240 return [&]<std::size_t... Is>(std::index_sequence<Is...>) {
241 return detail::bit_cast<R>(
242 __builtin_shufflevector(vec, VBuiltin{},
243 adj_idx(idx_perm2(vir::cw<Is>)).value...));
244 }(std::make_index_sequence<R::size()>());
245 }
246 }
247#endif
248 }
249#endif // __GNUC__
250
251 return R([&](auto i) -> T {
252 constexpr int j = [&] {
253 if constexpr (detail::index_permutation_function_nosize<F>)
254 return idx_perm(i);
255 else
256 return idx_perm(i, vir::cw<V::size()>);
257 }();
258 if constexpr (j == simd_permute_zero)
259 return 0;
260 else if constexpr (j == simd_permute_uninit)
261 {
262 T uninit;
263 return uninit;
264 }
265 else if constexpr (j < 0)
266 {
267 static_assert(-j <= int(V::size()));
268 return v[v.size() + j];
269 }
270 else
271 {
272 static_assert(j < int(V::size()));
273 return v[j];
274 }
275 });
276 }

279 template <std::size_t N = 0, vir::vectorizable T, detail::index_permutation_function<1> F>
280 VIR_ALWAYS_INLINE constexpr
281 std::conditional_t<N <= 1, T, stdx::resize_simd_t<N == 0 ? 1 : N, stdx::simd<T>>>
282 simd_permute(T const& v, F const idx_perm) noexcept
283 {
284 if constexpr (N <= 1)
285 {
286 constexpr auto i = vir::cw<0>;
287 constexpr int j = [&] {
288 if constexpr (detail::index_permutation_function_nosize<F>)
289 return idx_perm(i);
290 else
291 return idx_perm(i, vir::cw<std::size_t(1)>);
292 }();
293 if constexpr (j == simd_permute_zero)
294 return 0;
295 else if constexpr (j == simd_permute_uninit)
296 {
297 T uninit;
298 return uninit;
299 }
300 else
301 {
302 static_assert(j == 0 or j == -1);
303 return v;
304 }
305 }
306 else
307 return simd_permute<N>(stdx::simd<T, stdx::simd_abi::scalar>(v), idx_perm);
308 }

311 template <int Offset, vir::any_simd_or_mask V>
312 VIR_ALWAYS_INLINE constexpr V
313 simd_shift_in(V const& a, std::convertible_to<V> auto const&... more) noexcept
314 {
315 return V([&](auto i) -> typename V::value_type {
316 constexpr int ninputs = 1 + sizeof...(more);
317 constexpr int w = V::size();
318 constexpr int j = Offset + int(i);
319 if constexpr (j >= w * ninputs)
320 return 0;
321 else if constexpr (j >= 0)
322 {
323 const V tmp[] = {a, more...};
324 return tmp[j / w][j % w];
325 }
326 else if constexpr (j < -w)
327 return 0;
328 else
329 return a[w + j];
330 });
331 }
