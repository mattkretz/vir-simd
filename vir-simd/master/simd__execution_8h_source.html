48 // Invokes fun(V&) or fun(const V&) with V copied from ptr.
49 // If write_back is true, copy it back to ptr.
50 template <typename V, bool write_back = false, typename Flags = stdx::element_aligned_tag,
51 std::size_t... Is>
52 VIR_ALWAYS_INLINE constexpr void
53 simd_load_and_invoke(auto&& fun, auto ptr, Flags f, std::index_sequence<Is...>)
54 {
55 [&](auto... chunks) {
56 std::invoke(fun, chunks...);
57 if constexpr (write_back)
58 (chunks.copy_to(ptr + (V::size() * Is), f), ...);
59 }(std::conditional_t<write_back, V, const V>(ptr + (V::size() * Is), f)...);
60 }

62 template <typename V, typename Flags = stdx::element_aligned_tag, std::size_t... Is>
63 VIR_ALWAYS_INLINE constexpr void
64 simd_load_and_invoke(auto&& unary_op, auto ptr, auto&& result_op, Flags f, std::index_sequence<Is...>)
65 {
66 [&](const auto... chunks) {
67 (std::invoke(result_op, std::invoke(unary_op, chunks),
68 std::integral_constant<std::size_t, V::size() * Is>()), ...);
69 }(V(ptr + (V::size() * Is), f)...);

72 template <typename V1, typename V2, typename F1 = stdx::element_aligned_tag,
73 typename F2 = stdx::element_aligned_tag, std::size_t... Is>
74 VIR_ALWAYS_INLINE constexpr void
75 simd_load_and_invoke(auto&& binary_op, auto ptr1, auto ptr2, auto&& result_op, F1 f1, F2 f2,
76 std::index_sequence<Is...>)
77 {
78 constexpr auto size = V1::size();
79 static_assert(size == V2::size());
80 [&](const auto&... v1s) {
81 [&](const auto&... v2s) {
82 (std::invoke(result_op, std::invoke(binary_op, v1s, v2s),
83 std::integral_constant<std::size_t, size * Is>()), ...);
84 }(V2(ptr2 + (size * Is), f2)...);
85 }(V1(ptr1 + (size * Is), f1)...);
86 }

182 template <class V, bool write_back, std::size_t max_elements>
183 VIR_ALWAYS_INLINE constexpr void
184 simd_for_each_prologue(auto&& fun, auto ptr, std::size_t to_process)
185 {
186 static_assert(max_elements > 1);
187 static_assert(std::has_single_bit(max_elements));
188 static_assert(std::has_single_bit(unsigned(V::size())));
189 // pre-conditions:
190 // * ptr is a valid pointer to a range [ptr, ptr + to_process)
191 // * to_process > 0
192 // * to_process < max_elements
193
194 if (to_process == 0 or to_process >= max_elements)
195 unreachable();
196
197 if constexpr (max_elements == 2) // implies to_process == 1
198 {
199 simd_load_and_invoke<V, write_back>(fun, ptr, stdx::vector_aligned, no_unroll);
200 return;
201 }
202#if !__OPTIMIZE_SIZE__
203 if constexpr (V::size() == 1 and max_elements < 64)
204 if (not std::is_constant_evaluated())
205 return simd_for_each_jumptable_prologue<V, write_back, max_elements>(
206 fun, ptr, to_process);
207#endif
208
209 if (V::size() & to_process)
210 {
211 simd_load_and_invoke<V, write_back>(fun, ptr, stdx::vector_aligned, no_unroll);
212 ptr += V::size();
213 }
214 if constexpr (V::size() * 2 < max_elements)
215 {
216 simd_for_each_prologue<vir::simdize<typename V::value_type, V::size() * 2>,
217 write_back, max_elements>(fun, ptr, to_process);
218 }

314 template <class V0, bool write_back>
315 VIR_ALWAYS_INLINE constexpr void
316 simd_for_each_epilogue(auto&& fun, unsigned leftover, auto last, auto f)
317 {
318 // pre-conditions:
319 // * leftover > 0
320 // * leftover < V0::size()
321 // * [last - leftover, last) is a valid range
322 static_assert(V0::size() > 1);
323 using V1 = vir::simdize<typename V0::value_type, 1>;
324 if constexpr (V0::size() == 2) // implies leftover == 1
325 {
326 simd_load_and_invoke<V1, write_back>(fun, std::to_address(last - 1), f, no_unroll);
327 return;
328 }
329 if constexpr (V0::size() <= 4) // implies leftover ∈ [1, 3]
330 {
331 if (not std::is_constant_evaluated())
332 return simd_for_each_jumptable_epilogue<V0, write_back>(fun, leftover, last, f);
333 }
334#if !__OPTIMIZE_SIZE__
335 // The jumptable implementation for larger V0::size() can result in a lot of machine code.
336 // If the user wants to optimize for size then this isn't the right solution.
337 if constexpr (V0::size() <= 64
338 //and V0::size() <= vir::stdx::native_simd<typename V0::value_type>::size()
339 )
340 {
341 if (not std::is_constant_evaluated())
342 return simd_for_each_jumptable_epilogue<V0, write_back>(fun, leftover, last, f);
343 }
344#endif // !__OPTIMIZE_SIZE__
345 using V = vir::simdize<typename V0::value_type, std::bit_ceil(unsigned(V0::size())) / 2>;
346 if (leftover & V::size())
347 {
348 static_assert(std::has_single_bit(unsigned(V::size()))); // by construction
349 simd_load_and_invoke<V, write_back>(fun, std::to_address(last - leftover), f,
350 no_unroll);
351 leftover -= V::size();
352 }
353 if constexpr (V::size() > 1)
354 if (leftover)
355 simd_for_each_epilogue<V, write_back>(fun, leftover, last, f);
356 }

560 template <typename V, simd_execution_policy ExecutionPolicy>
561 struct prologue
562 {
563 using T = typename V::value_type;
564
565 static constexpr std::size_t size = V::size();
566
567 static constexpr auto bytes_per_iteration = size * sizeof(T);
568
569 static constexpr bool prologue_is_possible
570 = size > 1 and bytes_per_iteration % memory_alignment_v<V> == 0
571 and memory_alignment_v<V> > sizeof(T);
572
573 static constexpr bool use_aligned_loadstore
574 = ExecutionPolicy::_prefers_aligned and prologue_is_possible;
575
576 static constexpr bool maybe_execute_prologue_anyway
577 = ExecutionPolicy::_auto_prologue and prologue_is_possible;
578
579 static constexpr std::conditional_t<use_aligned_loadstore, stdx::vector_aligned_tag,
580 stdx::element_aligned_tag> flags{};
581
582 constexpr void
583 operator()(std::size_t& remaining, auto iterator_to_align, auto&& do_prologue,
584 auto&... iterators_to_advance) const
585 {
586 static_assert(std::same_as<T, std::iter_value_t<decltype(iterator_to_align)>>);
587
588 if constexpr (use_aligned_loadstore or maybe_execute_prologue_anyway)
589 {
590 constexpr auto max_misalignment = vir::cw<memory_alignment_v<V> / sizeof(T)>;
591 const auto misaligned_by_bytes
592 = reinterpret_cast<std::uintptr_t>(std::to_address(iterator_to_align))
593 % memory_alignment_v<V>;
594 if (misaligned_by_bytes == 0)
595 return;
596 const auto misaligned_by_elements = misaligned_by_bytes / sizeof(T);
597 const auto to_process = max_misalignment - misaligned_by_elements;
598 if constexpr (use_aligned_loadstore)
599 do_prologue(max_misalignment, to_process);
600 else if (remaining * sizeof(T) > 4000
601 or (to_process & remaining) == to_process) // prologue replaces epilogue
602 do_prologue(max_misalignment, to_process);
603 else
604 return;
605 ((iterators_to_advance += to_process), ...);
606 remaining -= to_process;
607 }
608 }

636 template <typename R, typename... Ts>
637 VIR_ALWAYS_INLINE constexpr void
638 simd_transform_prologue(auto&& op, auto dst_ptr, std::size_t to_process,
639 vir::constexpr_value auto max_elements, const Ts*... ptrs)
640 {
641 static_assert(max_elements > 1);
642 static_assert(std::has_single_bit(unsigned(max_elements)));
643 constexpr auto size = vir::cw<R::size()>;
644 static_assert(std::has_single_bit(unsigned(size)));
645 static_assert(size < max_elements);
646 // pre-conditions:
647 // * ptr is a valid pointer to a range [ptr, ptr + to_process)
648 // * to_process > 0
649 // * to_process < max_elements
650 // => if max_elements == 2 then to_process == 1 and size == 1
651
652 if (to_process == 0 or to_process >= max_elements)
653 unreachable();
654
655 if (max_elements == 2 or size & to_process)
656 {
657 const R& result = simdized_load_and_invoke(op, size, ptrs...);
658 result.copy_to(dst_ptr, stdx::vector_aligned);
659 if constexpr (max_elements == 2)
660 return;
661 ((ptrs += size), ...);
662 dst_ptr += size;
663 }
664 if constexpr (size * 2 < max_elements)
665 {
666 simd_transform_prologue<vir::resize_simdize_t<size * 2, R>, Ts...>(
667 op, dst_ptr, to_process, max_elements, ptrs...);
668 }
669 }

671 template <typename V0, typename R0, typename... More>
672 VIR_ALWAYS_INLINE constexpr void
673 simd_transform_epilogue(auto&& binary_op, unsigned leftover, auto last, auto d_first, auto f,
674 const More*... ptrs)
675 {
676 // pre-conditions:
677 // * leftover > 0
678 // * leftover < V0::size()
679 // * [last - leftover, last) is a valid range
680 // * [ptr2, ptr2 + leftover) is a valid range
681 // * [d_first, d_first + leftover) is a valid range
682 constexpr unsigned size0 = V0::size();
683 static_assert(size0 > 1);
684 static_assert(size0 == R0::size());
685 using T0 = typename V0::value_type;
686 if constexpr (size0 == 2) // implies leftover == 1
687 {
688 using R1 = vir::simdize<typename R0::value_type, 1>;
689 const R1& result = simdized_load_and_invoke(binary_op, 1_cw, last - 1, ptrs...);
690 result.copy_to(std::to_address(d_first), f);
691 return;
692 }
693 using V = vir::simdize<T0, std::bit_ceil(size0) / 2>;
694 using R = vir::simdize<typename R0::value_type, V::size()>;
695 if (leftover & V::size())
696 {
697 static_assert(std::has_single_bit(unsigned(V::size()))); // by construction
698 const R& result = simdized_load_and_invoke(binary_op, vir::cw<V::size()>,
699 last - leftover, ptrs...);
700 result.copy_to(std::to_address(d_first), f);
701 leftover -= V::size();
702 ((ptrs += V::size()), ...);
703 d_first += V::size();
704 }
705 if constexpr (V::size() > 1)
706 if (leftover)
707 simd_transform_epilogue<V, R, More...>(binary_op, leftover, last, d_first, f, ptrs...);
708 }

714 {
715 using T1 = std::iter_value_t<It1>;
716 using OutT = std::iter_value_t<OutIt>;
717 constexpr auto size = vir::cw<([] {
718 if constexpr (ExecutionPolicy::_size > 0)
719 return ExecutionPolicy::_size;
720 else
721 return std::max({int(iter_simdize_t<It1, 0>::size()),
722 int(iter_simdize_t<It2, 0>::size())...});
723 }())>;
724 using V1 = vir::simdize<T1, size>;
725 using OutV = vir::simdize<OutT, size>;
726
727 if (first1 == last1)
728 return d_first;
729
730 auto advance = [](int n, auto&... it) { ((it += n), ...); };
731
732 std::size_t distance = std::distance(first1, last1);
733 constexpr bool assume_matching_size = ExecutionPolicy::_assume_matching_size;
734 if constexpr (assume_matching_size)
735 vir_simd_precondition_vaargs(
736 distance % size == 0, "The explicit assumption, that the range size (%zu) is a multiple"
737 " of the SIMD width (%d), does not hold.", distance, size());
738
739 if (std::is_constant_evaluated())
740 {
741 // needs element_aligned because of GCC PR111302
742 for (; first1 + (size - 1) < last1; advance(size, first1, d_first, first2...))
743 {
744 const OutV& result = simdized_load_and_invoke(op, size, first1, first2...);
745 result.copy_to(std::to_address(d_first), stdx::element_aligned);
746 }
747
748 for (; first1 < last1; advance(1, first1, d_first, first2...))
749 {
750 const vir::simdize<OutT, 1>& result
751 = simdized_load_and_invoke(op, 1_cw, first1, first2...);
752 result.copy_to(std::to_address(d_first), stdx::element_aligned);
753 }
754 }
755 else
756 {
757 // There are three addresses and we can only ensure alignment on one of them. In
758 // principle, we could check at runtime whether two of three can be aligned. But the
759 // branching necessary to make this work - with relatively little gain (presumably) -
760 // renders this approach not interesting.
761 // Instead, let's simply focus on aligning the store address, avoiding pressure on the
762 // store execution ports (a factor of 2 throughput difference with AVX-512 vectors).
763 constexpr prologue<OutV, ExecutionPolicy> p;
764 constexpr auto flags = p.flags;
765 p(distance, d_first, [&] (auto max_elements, auto to_process) {
766 simd_transform_prologue<vir::simdize<OutT, 1>, T1, std::iter_value_t<It2>...>(
767 op, std::to_address(d_first), to_process, max_elements, std::to_address(first1),
768 std::to_address(first2)...);
769 }, first1, d_first, first2...);
770
771 if constexpr (ExecutionPolicy::_unroll_by > 1)
772 {
773 constexpr auto step = size * ExecutionPolicy::_unroll_by;
774 const auto unrolled_last = last1 - step;
775 for (; first1 <= unrolled_last; advance(step, first1, d_first, first2...))
776 {
777 unroll2<ExecutionPolicy::_unroll_by>([&,size](auto i) {
778 return simdized_load_and_invoke(op, size, first1 + i * size,
779 first2 + i * size...);
780 }, [&,size](auto i, const OutV& result) {
781 result.copy_to(std::to_address(d_first + i * size), flags);
782 });
783 }
784 }
785
786 const auto simd_last = last1 - size;
787 for (; assume_matching_size ? first1 != last1 : first1 <= simd_last;
788 advance(size, first1, d_first, first2...))
789 {
790 const OutV& result = simdized_load_and_invoke(op, size, first1, first2...);
791 result.copy_to(std::to_address(d_first), flags);
792 }
793
794 if constexpr (size > 1 and !assume_matching_size)
795 {
796 const auto leftover = distance % size;
797 if (leftover)
798 {
799 simd_transform_epilogue<V1, OutV, std::iter_value_t<It2>...>(
800 op, leftover, last1, d_first, flags, std::to_address(first2)...);
801 d_first += leftover;
802 }
803 }
804 }
805 return d_first;
806 }

808 template <int size, int max_size, typename A1, typename It1, typename... It2>
809 VIR_ALWAYS_INLINE constexpr A1
810 simd_transform_reduce_prologue(A1 acc1, It1 first1, std::size_t to_process,
811 auto reduce_op, auto transform_op, It2... first2)
812 {
813 // preconditions:
814 // - to_process > 0
815 // - to_process < max_size
816 // - [first1, first1 + to_process) is a valid range
817 // - [first2, first2 + to_process) is a valid range
818 static_assert(std::has_single_bit(unsigned(size)));
819 static_assert(std::has_single_bit(unsigned(max_size)));
820 if (to_process & size)
821 {
822 const std::size_t offset = to_process & (size - 1);
823 const A1 x = reduce(simdized_load_flag1_and_invoke(
824 transform_op, vir::cw<size>, stdx::vector_aligned,
825 first1 + offset, first2 + offset...),
826 reduce_op);
827 acc1 = std::invoke(reduce_op, acc1, x);
828 }
829 if constexpr (size < max_size)
830 return simd_transform_reduce_prologue<size * 2, max_size>(
831 acc1, first1, to_process, reduce_op, transform_op, first2...);
832 else
833 return acc1;
834 }

840 template <typename A1, typename A, typename It1, typename... It2>
841 VIR_ALWAYS_INLINE
842 constexpr typename A1::value_type
843 simd_transform_reduce_epilogue(A1 acc1, A acc, It1 first1, std::size_t leftover,
844 auto reduce_op, auto transform_op, auto flags1, It2... first2)
845 {
846 // preconditions:
847 // leftover > 0
848 // leftover & (A::size() * 2 - 1) != 0 (i.e. at least one more load+transform is necessary)
849 static_assert(A1::size() == 1);
850 static_assert(std::has_single_bit(unsigned(A::size())));
851 if (leftover & A::size())
852 {
853 acc = std::invoke(reduce_op, acc,
854 simdized_load_flag1_and_invoke(transform_op, vir::cw<A::size()>,
855 flags1, first1, first2...));
856 if constexpr (A::size() == 1)
857 // definetely no more work
858 return std::invoke(reduce_op, acc1, acc)[0];
859 else if ((leftover & (A::size() - 1)) == 0)
860 // no more work
861 return std::invoke(reduce_op, acc1, A1(reduce(acc, reduce_op)))[0];
862 else
863 {
864 constexpr std::size_t size2 = A::size() / 2;
865 using A2 = vir::simdize<typename A::value_type, size2>;
866 auto [lo, hi] = stdx::split<size2, size2>(acc);
867 A2 acc2 = std::invoke(reduce_op, lo, hi);
868 return simd_transform_reduce_epilogue(acc1, acc2, first1 + A::size(), leftover,
869 reduce_op, transform_op, flags1, first2 +
870 A::size()...);
871 }
872 }
873 else if ((leftover & (A::size() - 1)) == 0)
874 unreachable(); // because we're only called if there is more work to be done
875 else if constexpr (A::size() > 1)
876 {
877 constexpr std::size_t size2 = A::size() / 2;
878 using A2 = vir::simdize<typename A::value_type, size2>;
879 auto [lo, hi] = stdx::split<size2, size2>(acc);
880 A2 acc2 = std::invoke(reduce_op, lo, hi);
881 return simd_transform_reduce_epilogue(acc1, acc2, first1, leftover, reduce_op,
882 transform_op, flags1, first2...);
883 }
884 else
885 unreachable();
886 }

891 template <std::size_t size, typename A1, typename It1, typename... It2>
892 VIR_ALWAYS_INLINE
893 constexpr typename A1::value_type
894 simd_transform_reduce_epilogue(A1 acc1, It1 first1, std::size_t leftover,
895 auto reduce_op, auto transform_op, auto flags1, It2... first2)
896 {
897 // preconditions:
898 // leftover > 0
899 // leftover & (size * 2 - 1) != 0 (i.e. at least one more load+transform is necessary)
900 static_assert(A1::size() == 1);
901 static_assert(std::has_single_bit(size));
902 using A = vir::simdize<typename A1::value_type, size>;
903 if (leftover & size)
904 {
905 A acc = simdized_load_flag1_and_invoke(transform_op, vir::cw<size>, flags1, first1,
906 first2...);
907 if constexpr (size == 1)
908 // definetely no more work
909 return std::invoke(reduce_op, acc1, acc)[0];
910 else if ((leftover & (size - 1)) == 0)
911 // no more work
912 return std::invoke(reduce_op, acc1, A1(reduce(acc, reduce_op)))[0];
913 else
914 {
915 constexpr std::size_t size2 = size / 2;
916 using A2 = vir::simdize<typename A::value_type, size2>;
917 auto [lo, hi] = stdx::split<size2, size2>(acc);
918 A2 acc2 = std::invoke(reduce_op, lo, hi);
919 return simd_transform_reduce_epilogue(acc1, acc2, first1 + A::size(), leftover,
920 reduce_op, transform_op, flags1, first2 +
921 A::size()...);
922 }
923 }
924 else if ((leftover & (A::size() - 1)) == 0)
925 unreachable(); // because we're only called if there is more work to be done
926 else if constexpr (size > 1)
927 {
928 constexpr std::size_t size2 = A::size() / 2;
929 return simd_transform_reduce_epilogue<size2>(acc1, first1, leftover, reduce_op,
930 transform_op, flags1, first2...);
931 }
932 else
933 unreachable();
934 }

939 template <simd_execution_policy ExecutionPolicy, simd_execution_iterator It1, typename T,
940 typename BinaryReductionOp, typename TransformOp, simd_execution_iterator... It2>
941 constexpr T
942 transform_reduce(ExecutionPolicy, It1 first1, It1 last1, T init, BinaryReductionOp reduce_op,
943 TransformOp transform_op, It2... first2)
944 {
945 using T1 = std::iter_value_t<It1>;
946 constexpr int size = [] {
947 if constexpr (ExecutionPolicy::_size > 0)
948 return ExecutionPolicy::_size;
949 else
950 return std::max({int(iter_simdize_t<It1, 0>::size()),
951 int(iter_simdize_t<It2, 0>::size())...});
952 }();
953 using A1 = vir::simdize<T, 1>;
954 using A = vir::simdize<T, size>;
955 using V1 = vir::simdize<T1, size>;
956
957 if (first1 == last1)
958 return init;
959
960 A1 acc1 = init;
961
962 std::size_t distance = std::distance(first1, last1);
963 constexpr bool assume_matching_size = ExecutionPolicy::_assume_matching_size;
964 if constexpr (assume_matching_size)
965 vir_simd_precondition_vaargs(
966 distance % size == 0, "The explicit assumption, that the range size (%zu) is a multiple"
967 " of the SIMD width (%d), does not hold.", distance, size);
968
969 if (std::is_constant_evaluated())
970 {
971 // needs element_aligned because of GCC PR111302
972 for (; first1 < last1; ((first1 += 1), ..., (first2 += 1)))
973 {
974 const A1 r = simdized_load_and_invoke(transform_op, 1_cw, first1, first2...);
975 acc1 = std::invoke(reduce_op, acc1, r);
976 }
977 return acc1[0];
978 }
979 else
980 {
981 constexpr prologue<V1, ExecutionPolicy> p;
982 constexpr auto flags = p.flags;
983 p(distance, first1, [&] (vir::constexpr_value auto max_elements, auto to_process)
984 VIR_LAMBDA_ALWAYS_INLINE {
985 acc1 = simd_transform_reduce_prologue<1, max_elements>(
986 acc1, first1, to_process, reduce_op, transform_op, first2...);
987 }, first1, first2...);
988
989 const auto leftover = distance % size;
990
991 constexpr int lo_size = std::bit_ceil(unsigned(size)) / 2;
992 constexpr int hi_size = size - lo_size;
993
994 const auto simd_last = last1 - size;
995 // 'assume_matching_size' implies size is 0 or at least one simd width, and 0 has an
996 // early return above
997 if (assume_matching_size or first1 <= simd_last)
998 {
999 A acc = [&]() VIR_LAMBDA_ALWAYS_INLINE {
1000 if constexpr (ExecutionPolicy::_unroll_by > 1)
1001 {
1002 constexpr std::make_index_sequence<ExecutionPolicy::_unroll_by> unroll_idx_seq;
1003 constexpr auto step = size * ExecutionPolicy::_unroll_by;
1004 const auto unrolled_last = last1 - step;
1005 if (first1 + step <= unrolled_last)
1006 {
1007 auto acc = [&]<std::size_t... Is>(std::index_sequence<Is...>)
1008 -> std::array<A, ExecutionPolicy::_unroll_by>
1009 VIR_LAMBDA_ALWAYS_INLINE
1010 {
1011 return {
1012 [&](std::size_t offset) -> A VIR_LAMBDA_ALWAYS_INLINE {
1013 return simdized_load_flag1_and_invoke(
1014 transform_op, vir::cw<size>, flags,
1015 first1 + offset, first2 + offset...);
1016 }(Is * size)...
1017 };
1018 }(unroll_idx_seq);
1019 first1 += step;
1020 ((first2 += step), ...);
1021 do
1022 {
1023 [&]<std::size_t... Is>(std::index_sequence<Is...>)
1024 VIR_LAMBDA_ALWAYS_INLINE {
1025 ([&](std::size_t i) VIR_LAMBDA_ALWAYS_INLINE {
1026 acc[i] = std::invoke(
1027 reduce_op, acc[i],
1028 simdized_load_flag1_and_invoke(
1029 transform_op, vir::cw<size>, flags,
1030 first1 + i * size, first2 + i * size...));
1031 }(Is), ...);
1032 }(unroll_idx_seq);
1033 first1 += step;
1034 ((first2 += step), ...);
1035 }
1036 while (first1 <= unrolled_last);
1037 // tree reduction of acc into acc[0]
1038 unroll<std::bit_width(unsigned(ExecutionPolicy::_unroll_by - 1))>(
1039 [&](auto outer) {
1040 constexpr int j = 1 << outer.value; // j: 1 2 4 8 16 32 ...
1041 unroll<ExecutionPolicy::_unroll_by / 2>([&](auto ii) {
1042 constexpr int i = ii * 2 * j;
1043 if constexpr (i + j < ExecutionPolicy::_unroll_by)
1044 acc[i] = std::invoke(reduce_op, acc[i], acc[i+j]);
1045 });
1046 });
1047 return acc[0];
1048 }
1049 }
1050 auto ret = simdized_load_flag1_and_invoke(transform_op, vir::cw<size>, flags,
1051 first1, first2...);
1052 first1 += size;
1053 ((first2 += size), ...);
1054 return ret;
1055 }();
1056 for (; assume_matching_size ? first1 != last1 : first1 <= simd_last;
1057 ((first1 += size), ..., (first2 += size)))
1058 {
1059 acc = std::invoke(reduce_op, acc,
1060 simdized_load_flag1_and_invoke(transform_op, vir::cw<size>,
1061 flags, first1, first2...));
1062 }
1063 if constexpr (size == 1)
1064 return std::invoke(reduce_op, acc1, acc)[0];
1065 else if (assume_matching_size or leftover == 0)
1066 return std::invoke(reduce_op, acc1, A1(reduce(acc, reduce_op)))[0];
1067 else
1068 {
1069 auto [lo, hi] = stdx::split<lo_size, hi_size>(acc);
1070 vir::simdize<T, lo_size> acc2 = lo;
1071 if constexpr (lo_size == hi_size)
1072 acc2 = std::invoke(reduce_op, lo, hi);
1073 else
1074 acc1 = std::invoke(reduce_op, acc1, A1(reduce(hi, reduce_op)));
1075 return simd_transform_reduce_epilogue(acc1, acc2, first1, leftover, reduce_op,
1076 transform_op, flags, first2...);
1077 }
1078 }
1079 else if constexpr (lo_size > 0)
1080 {
1081 // leftover > 0 by construction
1082 return simd_transform_reduce_epilogue<lo_size>(
1083 acc1, first1, leftover, reduce_op, transform_op, flags, first2...);
1084 }
1085 else
1086 unreachable();
1087 }
1088 }

1110 template <detail::simd_execution_policy ExecutionPolicy, detail::simd_execution_iterator It,
1111 typename F>
1112 constexpr void
1113 for_each([[maybe_unused]] ExecutionPolicy pol, It first, It last, F&& fun)
1114 {
1115 using T = std::iter_value_t<It>;
1116 using V = vir::simdize<T, ExecutionPolicy::_size>;
1117 constexpr int size = V::size();
1118 constexpr bool write_back = std::indirectly_writable<It, T>
1119 and std::invocable<F, V&> and not std::invocable<F, V&&>;
1120 if (first == last)
1121 return;
1122
1123 std::size_t distance = std::distance(first, last);
1124 constexpr bool assume_matching_size = ExecutionPolicy::_assume_matching_size;
1125 if constexpr (assume_matching_size)
1126 vir_simd_precondition_vaargs(
1127 distance % size == 0, "The explicit assumption, that the range size (%zu) is a multiple"
1128 " of the SIMD width (%d), does not hold.", distance, size);
1129
1130 if (std::is_constant_evaluated())
1131 {
1132 // needs element_aligned because of GCC PR111302
1133 for (; first + (size - 1) < last; first += size)
1134 detail::simd_load_and_invoke<V, write_back>(fun, std::to_address(first),
1135 stdx::element_aligned, detail::no_unroll);
1136
1137 if constexpr (size > 1)
1138 detail::simd_for_each_epilogue<V, write_back>(fun, distance % size, last,
1139 stdx::element_aligned);
1140 }
1141 else
1142 {
1143 constexpr detail::prologue<V, ExecutionPolicy> prologue;
1144 constexpr auto flags = prologue.flags;
1145 prologue(distance, first, [&] (auto max_elements, auto to_process) {
1146 detail::simd_for_each_prologue<vir::simdize<T, 1>, write_back, max_elements>(
1147 fun, std::to_address(first), to_process);
1148 }, first);
1149 const auto leftover = distance % size;
1150
1151 if constexpr (ExecutionPolicy::_unroll_by > 1)
1152 {
1153 constexpr auto step = size * ExecutionPolicy::_unroll_by;
1154 const auto unrolled_last = last - step;
1155 for (; first <= unrolled_last; first += step)
1156 {
1157 detail::simd_load_and_invoke<V, write_back>(
1158 fun, std::to_address(first), flags,
1159 std::make_index_sequence<ExecutionPolicy::_unroll_by>());
1160 }
1161 }
1162
1163 const auto simd_last = last - size;
1164 for (; assume_matching_size ? first != last : first <= simd_last; first += size)
1165 detail::simd_load_and_invoke<V, write_back>(fun, std::to_address(first), flags,
1166 detail::no_unroll);
1167
1168 if constexpr (not assume_matching_size and size > 1)
1169 if (leftover)
1170 detail::simd_for_each_epilogue<V, write_back>(fun, leftover, last, flags);
1171 }
1172 }
